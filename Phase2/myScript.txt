#!/bin/bash

test01 > t1.out

for i in `seq 0 10`;
do 
	if [ "$i" -lt 10 ] 
	then 
		make test0$i
		test0$i > t0$i.out  # store outputs to a file
		diff t0$i.out ./testResults/test0$i.txt > diff0$i.out
	else
		make test$i
		test$i > t$i.out
		diff t0$i.out ./testResults/test0$i.txt > diff0$i.out
	fi
done
